xlrd和pymysql库需要简单了解一下

import xlrd,pymysql
//连接数据库
conn = pymysql.connect(host = '127.0.0.1',user = 'root',password = '123123',database = 'mysql',charset = 'utf8')
//创建游标
cursor = conn.cursor(cursor=pymysql.cursors.DictCursor)
sql = "insert into parasaga (UP_name,UP_UID,UP_fans_ranking,UP_fans_num,UP_play_ranking,UP_play_num,UP_point_ranking,UP_point_num) values (%s,%s,%s,%s,%s,%s,%s,%s)"
//进行excel读取数据
file = xlrd.open_workbook("D:/up_data.xls")
sheet_1 = file.sheet_by_index(0)
row_content = sheet_1.row_values(0)
row_number = sheet_1.nrows
for i in range(1,row_number):
    UP_name = sheet_1.cell(i,0).value
    UP_UID = sheet_1.cell(i,1).value
    UP_fans_ranking = sheet_1.cell(i,2).value
    UP_fans_num = sheet_1.cell(i,3).value
    UP_play_ranking = sheet_1.cell(i,4).value
    UP_play_num = sheet_1.cell(i,5).value
    UP_point_ranking = sheet_1.cell(i,6).value
    UP_point_num = sheet_1.cell(i,7).value
    values = (UP_name,UP_UID,UP_fans_ranking,UP_fans_num,UP_play_ranking,UP_play_num,UP_point_ranking,UP_point_num)
    //执行sql语句
    cursor.execute(sql,values)
    conn.commit()
cursor.close()
conn.close()
